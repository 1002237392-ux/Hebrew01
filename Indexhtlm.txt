<script>
const correctAnswers = { s1:'direct', s2:'indirect', s3:'direct', s4:'indirect', s5:'direct', s6:'indirect' };
let draggedItem = null;

document.querySelectorAll('.draggable-sentence').forEach(el => {
  el.addEventListener('dragstart', () => { draggedItem = el; setTimeout(() => el.classList.add('dragging'), 0); });
  el.addEventListener('dragend', () => { el.classList.remove('dragging'); draggedItem = null; });
});

document.querySelectorAll('.drop-zone').forEach(zone => {
  zone.addEventListener('dragover', e => { e.preventDefault(); zone.classList.add('drag-over'); });
  zone.addEventListener('dragleave', () => zone.classList.remove('drag-over'));
  zone.addEventListener('drop', e => {
    e.preventDefault();
    zone.classList.remove('drag-over');
    if (!draggedItem) return;
    const id = draggedItem.dataset.id;
    const text = draggedItem.textContent.trim();
    const zoneId = zone.id === 'directZone' ? 'direct' : 'indirect';
    const existing = document.querySelector(`.dropped-item[data-id="${id}"]`);
    if (existing) existing.remove();
    draggedItem.style.display = 'none';
    const item = document.createElement('div');
    item.className = 'dropped-item';
    item.dataset.id = id;
    item.dataset.zone = zoneId;
    item.innerHTML = `<span style="font-family:'Frank Ruhl Libre',serif;">${text}</span>
      <button class="remove-btn" onclick="returnToPool('${id}')">✕</button>`;
    zone.querySelector('.dropped-items').appendChild(item);
    updateProgress();
  });
});

function returnToPool(id) {
  const item = document.querySelector(`.dropped-item[data-id="${id}"]`);
  if (item) item.remove();
  const orig = document.querySelector(`.draggable-sentence[data-id="${id}"]`);
  if (orig) orig.style.display = '';
  updateProgress();
}

function checkDragDrop() {
  const placed = document.querySelectorAll('.dropped-item');
  if (!placed.length) { showResult('dndResult','partial','⚠️ טרם הנחת משפטים! גרור/י את המשפטים לתיבות.'); return; }
  let correct = 0;
  placed.forEach(item => { if (correctAnswers[item.dataset.id] === item.dataset.zone) correct++; });
  const total = Object.keys(correctAnswers).length;
  if (correct === total) showResult('dndResult','success',`🎉 מצוין! כל התשובות נכונות! (${correct}/${total})`);
  else if (correct >= Math.ceil(total/2)) showResult('dndResult','partial',`👍 טוב! ${correct} מתוך ${total} נכונות. בדוק/י שוב את השגיאות.`);
  else showResult('dndResult','wrong',`💪 ${correct} מתוך ${total} נכונות. חזור/י על המושגים ונסה/י שוב!`);
}

function resetDragDrop() {
  document.querySelectorAll('.dropped-item').forEach(el => el.remove());
  document.querySelectorAll('.draggable-sentence').forEach(el => el.style.display = '');
  document.getElementById('dndResult').classList.remove('show');
  updateProgress();
}

function showResult(id, type, msg) {
  const el = document.getElementById(id);
  el.className = `check-result ${type} show`;
  el.textContent = msg;
}

function setStars(groupId, count) {
  const stars = document.querySelectorAll(`#${groupId} .star`);
  const total = stars.length;
  stars.forEach((s,i) => (total - i) <= count ? s.classList.add('active') : s.classList.remove('active'));
  updateProgress();
}

function updateProgress() {
  const ids = ['q2_1','q2_2','q2_3','q2_4','t1','t2','t3','t4','t5','q4_creative','ref1','ref2','ref3','ref4','ref_final'];
  let filled = ids.filter(id => { const el = document.getElementById(id); return el && el.value.trim().length > 3; }).length;
  if (document.querySelectorAll('.dropped-item').length > 0) filled += 2;
  const total = ids.length + 2;
  const pct = Math.round((filled / total) * 100);
  document.getElementById('progressBar').style.width = pct + '%';
  document.getElementById('progressScore').textContent = pct + ' / 100';
}
